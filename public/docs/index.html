POST api/phone/verify/hook

Example request:
curl --request POST \
    "http://localhost:8000/api/phone/verify/hook" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json" \
    --data "{
    \"key\": \"voluptas\",
    \"phone\": \"praesentium\"
}"

const url = new URL(
    "http://localhost:8000/api/phone/verify/hook"
);

const headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

let body = {
    "key": "voluptas",
    "phone": "praesentium"
};

fetch(url, {
    method: "POST",
    headers,
    body: JSON.stringify(body),
}).then(response => response.json());
