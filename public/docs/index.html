Introduction

Base URL: http://localhost:8000

This documentation aims to provide all the information you need to work with our API.

As you scroll, you'll see code examples for working with the API in different programming languages in the dark area to the right (or as part of the content on mobile). You can switch the language used with the tabs at the top right (or from the nav menu at the top left on mobile).

Authenticating requests

This API is not authenticated.

Auth

Авторизация пользователей и всё что с ней связано.

Login

Авторизация пользователя

Example request:
curl --request POST \
    "http://localhost:8000/api/login" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json" \
    --data "{
    \"email\": \"hadley.kuhic@example.org\",
    \"password\": \"k\'Ng}<_Bvu1DuAeH\"
}"

const url = new URL(
    "http://localhost:8000/api/login"
);

const headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

let body = {
    "email": "hadley.kuhic@example.org",
    "password": "k'Ng}<_Bvu1DuAeH"
};

fetch(url, {
    method: "POST",
    headers,
    body: JSON.stringify(body),
}).then(response => response.json());

create

requires authentication

Метод создания трека

Example request:
curl --request POST \
    "http://localhost:8000/api/tracks" \
    --header "Content-Type: multipart/form-data" \
    --header "Accept: application/json" \
    --form "name=uzhgrxqbpyhjimxxpmodwrtftfvmwlkqfremywakhicmynz"\
    --form "address=minima"\
    --form "latitude=-86"\
    --form "longitude=-179"\
    --form "levelId=10"\
    --form "desc=vel"\
    --form "length=6"\
    --form "turns=19"\
    --form "free=1"\
    --form "is_work="\
    --form "spec=["omnis","aut"]"\
    --form "images[]=@/tmp/phpdkhlFn" 
const url = new URL(
    "http://localhost:8000/api/tracks"
);

const headers = {
    "Content-Type": "multipart/form-data",
    "Accept": "application/json",
};

const body = new FormData();
body.append('name', 'uzhgrxqbpyhjimxxpmodwrtftfvmwlkqfremywakhicmynz');
body.append('address', 'minima');
body.append('latitude', '-86');
body.append('longitude', '-179');
body.append('levelId', '10');
body.append('desc', 'vel');
body.append('length', '6');
body.append('turns', '19');
body.append('free', '1');
body.append('is_work', '');
body.append('spec', '["omnis","aut"]');
body.append('images[]', document.querySelector('input[name="images[]"]').files[0]);

fetch(url, {
    method: "POST",
    headers,
    body,
}).then(response => response.json());
