const url = new URL(
    "http://localhost/api/login"
);

const headers = {
    "Content-Type": "application/json",

};

    "email": "jaunita.swaniawski@example.org",
    "password": "|g][c!;vH5"

    method: "POST",
    headers,
    body: JSON.stringify(body),
